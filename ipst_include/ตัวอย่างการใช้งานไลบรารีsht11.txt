การใช้งานไลบรารี  sht11_lib.h
 หลังจากทำการผนวกไลบรารีเรียบร้อยแล้ว


ตัวอย่างนะครับ
Example 1

#include <sht11_lib.h>

void main()
{
   int temp,humi;
   ...........
   temp = sht11_temp();	 // อ่านค่าอุณหภูมิแบบค่าจำนวนเต็ม
   humi = sht11_humi();  // อ่านค่าความชื้นสัมพัทแบบจำนวนเต็ม
   ...........

}
